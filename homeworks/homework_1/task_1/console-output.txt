myName: String = Vlad

scala> val Greetings = List("Hello", "Hola", "Guten tag")
Greetings: List[String] = List(Hello, Hola, Guten tag)

scala> for (greeting <- Greetings) println(f"${greeting} Scala! This is ${myName}")

Hola Scala! This is Vlad
Guten tag Scala! This is Vlad

scala> for (greeting <- Greetings) println(f"${greeting} Scale! This is ${myName.reverse}")

Guten tag Scale! This is dalV
